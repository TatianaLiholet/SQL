
CREATE TABLE public.wallet
(
    wallet_id integer NOT NULL,
    currency text,
    amount integer,
    user_id integer NOT NULL,
    " created" date,
    PRIMARY KEY (wallet_id)
);

CREATE TABLE public."user"
(
    user_id integer NOT NULL,
    name text,
    "surname " text,
    "date_of_registration " date,
    PRIMARY KEY (user_id)
);
